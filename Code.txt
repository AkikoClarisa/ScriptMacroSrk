using System;
using System.Windows.Forms;
using System.Threading;
using System.Runtime.InteropServices;

public class Program
{
    [DllImport("user32.dll")]
    static extern bool SetCursorPos(int X, int Y);

    [DllImport("user32.dll", CharSet = CharSet.Auto, CallingConvention = CallingConvention.StdCall)]
    public static extern void mouse_event(uint dwFlags, uint dx, uint dy, uint cButtons, uint dwExtraInfo);

    private const int MOUSEEVENTF_LEFTDOWN = 0x02;
    private const int MOUSEEVENTF_LEFTUP = 0x04;

    public static void KlikMouse(int x, int y)
    {
        SetCursorPos(x, y); 
        Thread.Sleep(100); 
        mouse_event(MOUSEEVENTF_LEFTDOWN | MOUSEEVENTF_LEFTUP, 0, 0, 0, 0); 
        Thread.Sleep(300); 
    }

    public static void Main()
    {
        // =======================================================
        // KORDINAT FULL (Udah dimasukin semua)
        // =======================================================
        int excelAmanX = 366; int excelAmanY = 28;     // Area hijau atas Excel
        
        int webNikX = 1405; int webNikY = 391;         // Kolom NIK Web
        int webTglX = 1383; int webTglY = 469;         // Kolom Tgl Lahir Web
        int webCaptchaX = 1334; int webCaptchaY = 604; // Kolom Captcha
        int btnCariX = 1591; int btnCariY = 543;       // Tombol Cari Peserta
        // =======================================================

        // 1. Copy NIK (Posisikan lu udah klik 1 NIK yang putih di Excel sblm di-play)
        SendKeys.SendWait("^c"); 
        Thread.Sleep(300);

        // 2. Paste NIK ke Web
        KlikMouse(webNikX, webNikY);
        SendKeys.SendWait("^v");
        Thread.Sleep(300);

        // 3. Aktifin Excel, geser KANAN 4 KALI ke Tanggal Lahir & Copy
        KlikMouse(excelAmanX, excelAmanY);
        SendKeys.SendWait("{RIGHT 4}"); 
        Thread.Sleep(300);
        SendKeys.SendWait("^c");
        Thread.Sleep(300);

        // 4. Paste Tanggal Lahir ke Web
        KlikMouse(webTglX, webTglY);
        SendKeys.SendWait("^v");
        Thread.Sleep(300);

        // 5. Aktifin Excel lagi, balikin kursor ke kolom NIK awal (Geser KIRI 4 KALI)
        KlikMouse(excelAmanX, excelAmanY);
        SendKeys.SendWait("{LEFT 4}"); 
        Thread.Sleep(100);

        // 6. Klik kolom Captcha di web biar kursornya kelap-kelip siap diketik
        KlikMouse(webCaptchaX, webCaptchaY);

        // 7. Pause pake MessageBox (Tunggu lu ngetik Captcha)
        MessageBox.Show("Silakan ketik Captcha di web sekarang.\nKalau sudah selesai diketik, klik OK di kotak ini.", "Waktunya Isi Captcha!", MessageBoxButtons.OK, MessageBoxIcon.Information);

        // 8. Setelah lu klik OK, script otomatis ngeklik "Cari Peserta"
        KlikMouse(btnCariX, btnCariY);
        
        // Jeda 3 detik nunggu web loading
        Thread.Sleep(3000); 
    }
}
